# Workload Transform Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build an AI-powered workload transform system that takes a captured `.wkl` profile + natural language prompt, generates a structured transform plan, and produces a modified workload for replay.

**Architecture:** Three-layer design: (1) deterministic Analyzer extracts table names and groups queries, (2) AI Planner interprets user prompt and generates a TOML TransformPlan, (3) deterministic Engine applies the plan to produce a new `.wkl` file. Multi-provider LLM support (Claude, OpenAI, Ollama).

**Tech Stack:** Rust, regex (table extraction), reqwest (LLM HTTP), toml/serde (plan format), rand (seeded RNG for injection), existing profile/replay/transform infrastructure.

```rust
use serde::{Deserialize, Serialize};

/// A structured plan for transforming a workload profile.
/// Generated by AI, editable by human, executed by deterministic engine.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TransformPlan {
    pub version: u8,
    pub source: PlanSource,
    pub analysis: PlanAnalysis,
    pub groups: Vec<QueryGroup>,
    pub transforms: Vec<TransformRule>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct PlanSource {
    pub profile: String,
    pub prompt: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct PlanAnalysis {
    pub total_queries: u64,
    pub total_sessions: u64,
    pub groups_identified: usize,
}

/// A group of queries identified by shared table access patterns.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct QueryGroup {
    pub name: String,
    pub description: String,
    pub tables: Vec<String>,
    pub query_indices: Vec<usize>,
    pub session_ids: Vec<u64>,
    pub query_count: usize,
}

/// A transform rule to apply to the workload.
#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(tag = "type", rename_all = "snake_case")]
pub enum TransformRule {
    Scale {
        group: String,
        factor: f64,
        #[serde(default = "default_stagger")]
        stagger_ms: u64,
    },
    Inject {
        description: String,
        sql: String,
        after_group: String,
        #[serde(default = "default_frequency")]
        frequency: f64,
        #[serde(default = "default_duration")]
        estimated_duration_us: u64,
    },
    InjectSession {
        description: String,
        queries: Vec<InjectedQuery>,
        repeat: u32,
        #[serde(default = "default_interval")]
        interval_ms: u64,
    },
    Remove {
        group: String,
    },
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct InjectedQuery {
    pub sql: String,
    pub duration_us: u64,
}

fn default_stagger() -> u64 { 10 }
fn default_frequency() -> f64 { 1.0 }
fn default_duration() -> u64 { 1000 }
fn default_interval() -> u64 { 1000 }
```

```rust
use std::collections::{BTreeSet, HashMap, HashSet};
use regex::Regex;
use serde::Serialize;

use crate::profile::{QueryKind, WorkloadProfile};

/// Extract table names from a SQL statement using regex patterns.
pub fn extract_tables(sql: &str) -> BTreeSet<String> {
    let mut tables = BTreeSet::new();
    let sql_upper = sql.to_uppercase();

    // Patterns: FROM table, JOIN table, INTO table, UPDATE table, DELETE FROM table
    let patterns = [
        r"(?i)\bFROM\s+([a-zA-Z_][a-zA-Z0-9_.]*)",
        r"(?i)\bJOIN\s+([a-zA-Z_][a-zA-Z0-9_.]*)",
        r"(?i)\bINTO\s+([a-zA-Z_][a-zA-Z0-9_.]*)",
        r"(?i)\bUPDATE\s+([a-zA-Z_][a-zA-Z0-9_.]*)",
    ];

    for pattern in &patterns {
        let re = Regex::new(pattern).unwrap();
        for cap in re.captures_iter(sql) {
            if let Some(m) = cap.get(1) {
                let table = m.as_str().to_lowercase();
                // Skip SQL keywords that look like table names
                if !is_sql_keyword(&table) {
                    tables.insert(table);
                }
            }
        }
    }

    tables
}

/// Extract WHERE clause column names.
pub fn extract_filter_columns(sql: &str) -> Vec<String> {
    let mut cols = Vec::new();
    let re = Regex::new(r"(?i)\bWHERE\b(.+)").unwrap();
    if let Some(cap) = re.captures(sql) {
        let where_clause = &cap[1];
        let col_re = Regex::new(r"(?i)\b([a-zA-Z_][a-zA-Z0-9_]*)\s*(?:=|!=|<>|>=|<=|>|<|IN|LIKE|IS|BETWEEN)").unwrap();
        for cap in col_re.captures_iter(where_clause) {
            if let Some(m) = cap.get(1) {
                let col = m.as_str().to_lowercase();
                if !is_sql_keyword(&col) && !cols.contains(&col) {
                    cols.push(col);
                }
            }
        }
    }
    cols
}

fn is_sql_keyword(word: &str) -> bool {
    matches!(
        word,
        "select" | "from" | "where" | "and" | "or" | "not" | "in" | "is"
        | "null" | "true" | "false" | "as" | "on" | "set" | "values"
        | "order" | "by" | "group" | "having" | "limit" | "offset"
        | "union" | "all" | "exists" | "between" | "like" | "case"
        | "when" | "then" | "else" | "end" | "inner" | "outer"
        | "left" | "right" | "cross" | "natural" | "using"
    )
}
```

```rust
/// Structured analysis of a workload for AI planner consumption.
#[derive(Debug, Clone, Serialize)]
pub struct WorkloadAnalysis {
    pub profile_summary: ProfileSummary,
    pub query_groups: Vec<AnalyzedGroup>,
    pub ungrouped_queries: usize,
}

#[derive(Debug, Clone, Serialize)]
pub struct ProfileSummary {
    pub total_queries: u64,
    pub total_sessions: u64,
    pub capture_duration_s: f64,
    pub source_host: String,
}

#[derive(Debug, Clone, Serialize)]
pub struct AnalyzedGroup {
    pub id: usize,
    pub tables: Vec<String>,
    pub query_count: usize,
    pub sample_queries: Vec<String>,
    pub kinds: HashMap<String, usize>,
    pub avg_duration_us: u64,
    pub sessions: Vec<u64>,
    pub pct_of_total: f64,
    pub parameter_patterns: ParameterPatterns,
}

#[derive(Debug, Clone, Serialize)]
pub struct ParameterPatterns {
    pub common_filters: Vec<String>,
    pub bind_params_seen: usize,
}

/// Analyze a workload profile: extract tables, group queries, compute stats.
pub fn analyze_workload(profile: &WorkloadProfile) -> WorkloadAnalysis {
    // Step 1: Extract tables for every query (flattened across sessions)
    struct QueryInfo {
        tables: BTreeSet<String>,
        sql: String,
        kind: QueryKind,
        duration_us: u64,
        session_id: u64,
        filters: Vec<String>,
    }

    let mut all_queries: Vec<QueryInfo> = Vec::new();
    for session in &profile.sessions {
        for query in &session.queries {
            all_queries.push(QueryInfo {
                tables: extract_tables(&query.sql),
                sql: query.sql.clone(),
                kind: query.kind,
                duration_us: query.duration_us,
                session_id: session.id,
                filters: extract_filter_columns(&query.sql),
            });
        }
    }

    // Step 2: Union-Find to merge tables that appear together
    let mut table_to_group: HashMap<String, usize> = HashMap::new();
    let mut parent: Vec<usize> = Vec::new();

    fn find(parent: &mut Vec<usize>, mut i: usize) -> usize {
        while parent[i] != i {
            parent[i] = parent[parent[i]]; // path compression
            i = parent[i];
        }
        i
    }

    fn union(parent: &mut Vec<usize>, a: usize, b: usize) {
        let ra = find(parent, a);
        let rb = find(parent, b);
        if ra != rb {
            parent[ra] = rb;
        }
    }

    for qi in &all_queries {
        let tables: Vec<&String> = qi.tables.iter().collect();
        if tables.is_empty() {
            continue;
        }
        // Ensure all tables have a group ID
        for table in &tables {
            if !table_to_group.contains_key(table.as_str()) {
                let id = parent.len();
                parent.push(id);
                table_to_group.insert(table.to_string(), id);
            }
        }
        // Union all tables in this query together
        let first = table_to_group[tables[0]];
        for table in &tables[1..] {
            let other = table_to_group[table.as_str()];
            union(&mut parent, first, other);
        }
    }

    // Step 3: Collect queries into groups by their root table group
    let mut group_queries: HashMap<usize, Vec<usize>> = HashMap::new();
    let mut ungrouped = 0usize;

    for (idx, qi) in all_queries.iter().enumerate() {
        if qi.tables.is_empty() {
            ungrouped += 1;
            continue;
        }
        let first_table = qi.tables.iter().next().unwrap();
        let group_id = find(&mut parent, table_to_group[first_table]);
        group_queries.entry(group_id).or_default().push(idx);
    }

    // Step 4: Build AnalyzedGroup for each group
    let mut groups: Vec<AnalyzedGroup> = Vec::new();
    let total_queries = all_queries.len();

    for (group_idx, (_root, query_indices)) in group_queries.iter().enumerate() {
        let mut tables_set: BTreeSet<String> = BTreeSet::new();
        let mut kinds: HashMap<String, usize> = HashMap::new();
        let mut sessions: HashSet<u64> = HashSet::new();
        let mut total_duration: u64 = 0;
        let mut samples: Vec<String> = Vec::new();
        let mut all_filters: Vec<String> = Vec::new();
        let mut bind_count = 0usize;

        for &idx in query_indices {
            let qi = &all_queries[idx];
            tables_set.extend(qi.tables.iter().cloned());
            *kinds.entry(format!("{:?}", qi.kind)).or_default() += 1;
            sessions.insert(qi.session_id);
            total_duration += qi.duration_us;
            if samples.len() < 5 {
                let preview = if qi.sql.len() > 120 {
                    format!("{}...", &qi.sql[..117])
                } else {
                    qi.sql.clone()
                };
                if !samples.contains(&preview) {
                    samples.push(preview);
                }
            }
            for f in &qi.filters {
                if !all_filters.contains(f) {
                    all_filters.push(f.clone());
                }
            }
            // Count bind parameters
            let bind_re = Regex::new(r"\$\d+").unwrap();
            bind_count += bind_re.find_iter(&qi.sql).count();
        }

        let count = query_indices.len();
        let mut session_ids: Vec<u64> = sessions.into_iter().collect();
        session_ids.sort();

        groups.push(AnalyzedGroup {
            id: group_idx,
            tables: tables_set.into_iter().collect(),
            query_count: count,
            sample_queries: samples,
            kinds,
            avg_duration_us: if count > 0 { total_duration / count as u64 } else { 0 },
            sessions: session_ids,
            pct_of_total: if total_queries > 0 { (count as f64 / total_queries as f64) * 100.0 } else { 0.0 },
            parameter_patterns: ParameterPatterns {
                common_filters: all_filters,
                bind_params_seen: bind_count,
            },
        });
    }

    // Sort groups by query count descending
    groups.sort_by(|a, b| b.query_count.cmp(&a.query_count));
    // Reassign IDs after sort
    for (i, g) in groups.iter_mut().enumerate() {
        g.id = i;
    }

    WorkloadAnalysis {
        profile_summary: ProfileSummary {
            total_queries: profile.metadata.total_queries,
            total_sessions: profile.metadata.total_sessions,
            capture_duration_s: profile.metadata.capture_duration_us as f64 / 1_000_000.0,
            source_host: profile.source_host.clone(),
        },
        query_groups: groups,
        ungrouped_queries: ungrouped,
    }
}
```

```rust
use std::collections::{HashMap, HashSet};

use anyhow::{bail, Result};
use rand::rngs::StdRng;
use rand::{Rng, SeedableRng};

use crate::profile::{Metadata, Query, QueryKind, Session, WorkloadProfile};
use super::plan::{TransformPlan, TransformRule, QueryGroup};

/// Apply a transform plan to a workload profile, producing a new profile.
///
/// `seed` controls RNG for frequency-based injection. If None, uses hash of plan.
pub fn apply_transform(
    profile: &WorkloadProfile,
    plan: &TransformPlan,
    seed: Option<u64>,
) -> Result<WorkloadProfile> {
    let seed = seed.unwrap_or_else(|| {
        use std::hash::{Hash, Hasher};
        let mut hasher = std::collections::hash_map::DefaultHasher::new();
        plan.source.prompt.hash(&mut hasher);
        hasher.finish()
    });
    let mut rng = StdRng::seed_from_u64(seed);

    // Build flat index → (session_idx, query_idx) mapping
    let mut flat_to_local: Vec<(usize, usize)> = Vec::new();
    for (si, session) in profile.sessions.iter().enumerate() {
        for qi in 0..session.queries.len() {
            flat_to_local.push((si, qi));
        }
    }

    // Build group membership: flat_index → group_name
    let mut query_group: HashMap<usize, String> = HashMap::new();
    for group in &plan.groups {
        for &idx in &group.query_indices {
            query_group.insert(idx, group.name.clone());
        }
    }

    // Collect transform rules by type
    let mut scale_rules: HashMap<String, f64> = HashMap::new();
    let mut stagger_rules: HashMap<String, u64> = HashMap::new();
    let mut remove_groups: HashSet<String> = HashSet::new();
    let mut inject_rules = Vec::new();
    let mut inject_session_rules = Vec::new();

    for rule in &plan.transforms {
        match rule {
            TransformRule::Scale { group, factor, stagger_ms } => {
                scale_rules.insert(group.clone(), *factor);
                stagger_rules.insert(group.clone(), *stagger_ms);
            }
            TransformRule::Remove { group } => {
                remove_groups.insert(group.clone());
            }
            TransformRule::Inject { .. } => {
                inject_rules.push(rule.clone());
            }
            TransformRule::InjectSession { .. } => {
                inject_session_rules.push(rule.clone());
            }
        }
    }

    // Step 1: Build modified sessions (remove + inject into existing sessions)
    let mut sessions: Vec<Session> = Vec::new();
    let mut flat_idx = 0usize;

    for session in &profile.sessions {
        let mut new_queries: Vec<Query> = Vec::new();

        for query in &session.queries {
            let group_name = query_group.get(&flat_idx);

            // Skip removed queries
            if let Some(gn) = group_name {
                if remove_groups.contains(gn) {
                    flat_idx += 1;
                    continue;
                }
            }

            new_queries.push(query.clone());

            // Check inject rules
            if let Some(gn) = group_name {
                for rule in &inject_rules {
                    if let TransformRule::Inject { sql, after_group, frequency, estimated_duration_us, .. } = rule {
                        if after_group == gn && rng.gen::<f64>() < *frequency {
                            let injected = Query {
                                sql: sql.clone(),
                                start_offset_us: query.start_offset_us + query.duration_us,
                                duration_us: *estimated_duration_us,
                                kind: QueryKind::from_sql(sql),
                                transaction_id: None,
                            };
                            new_queries.push(injected);
                        }
                    }
                }
            }

            flat_idx += 1;
        }

        if !new_queries.is_empty() {
            sessions.push(Session {
                id: session.id,
                user: session.user.clone(),
                database: session.database.clone(),
                queries: new_queries,
            });
        }
    }

    // Step 2: Weighted session duplication for scale rules
    let mut scaled_sessions: Vec<Session> = Vec::new();
    let max_id = sessions.iter().map(|s| s.id).max().unwrap_or(0);
    let mut next_id = max_id + 1;
    let mut global_stagger_count: u64 = 0;

    for session in &sessions {
        // Always keep the original
        scaled_sessions.push(session.clone());

        // Compute weighted scale factor for this session
        let mut group_query_counts: HashMap<String, usize> = HashMap::new();
        // Re-analyze which groups this session's queries belong to
        // (simplified: check sql against group tables)
        let total_q = session.queries.len() as f64;
        if total_q == 0.0 {
            continue;
        }

        // Count queries per group by checking original flat indices
        // We need to reconstruct which group each query belongs to
        for group in &plan.groups {
            let count = session.queries.iter().filter(|q| {
                let tables = super::analyze::extract_tables(&q.sql);
                group.tables.iter().any(|t| tables.contains(t))
            }).count();
            if count > 0 {
                group_query_counts.insert(group.name.clone(), count);
            }
        }

        let mut weighted_scale = 0.0f64;
        let mut accounted = 0usize;
        for (gn, count) in &group_query_counts {
            let factor = scale_rules.get(gn).copied().unwrap_or(1.0);
            weighted_scale += (*count as f64 / total_q) * factor;
            accounted += count;
        }
        // Unaccounted queries treated as scale 1.0
        let unaccounted = total_q as usize - accounted;
        if unaccounted > 0 {
            weighted_scale += (unaccounted as f64 / total_q) * 1.0;
        }

        let copies = (weighted_scale.round() as u64).max(1) - 1; // -1 because original already added
        let stagger_us = stagger_rules.values().copied().max().unwrap_or(10) * 1000;

        for _ in 0..copies {
            global_stagger_count += 1;
            let offset = global_stagger_count * stagger_us;
            let new_session = Session {
                id: next_id,
                user: session.user.clone(),
                database: session.database.clone(),
                queries: session.queries.iter().map(|q| {
                    Query {
                        sql: q.sql.clone(),
                        start_offset_us: q.start_offset_us + offset,
                        duration_us: q.duration_us,
                        kind: q.kind,
                        transaction_id: q.transaction_id,
                    }
                }).collect(),
            };
            scaled_sessions.push(new_session);
            next_id += 1;
        }
    }

    // Step 3: Inject new sessions
    let capture_duration_us = profile.metadata.capture_duration_us;
    for rule in &inject_session_rules {
        if let TransformRule::InjectSession { queries, repeat, interval_ms, .. } = rule {
            let interval_us = *interval_ms * 1000;
            for i in 0..*repeat {
                let base_offset = (i as u64) * interval_us;
                let new_queries: Vec<Query> = queries.iter().enumerate().map(|(qi, iq)| {
                    Query {
                        sql: iq.sql.clone(),
                        start_offset_us: base_offset + (qi as u64 * iq.duration_us),
                        duration_us: iq.duration_us,
                        kind: QueryKind::from_sql(&iq.sql),
                        transaction_id: None,
                    }
                }).collect();

                scaled_sessions.push(Session {
                    id: next_id,
                    user: profile.sessions.first().map(|s| s.user.clone()).unwrap_or_else(|| "transform".into()),
                    database: profile.sessions.first().map(|s| s.database.clone()).unwrap_or_else(|| "postgres".into()),
                    queries: new_queries,
                });
                next_id += 1;
            }
        }
    }

    // Step 4: Update metadata
    let total_queries: u64 = scaled_sessions.iter().map(|s| s.queries.len() as u64).sum();
    let total_sessions = scaled_sessions.len() as u64;
    let max_offset = scaled_sessions.iter().flat_map(|s| s.queries.iter()).map(|q| q.start_offset_us + q.duration_us).max().unwrap_or(capture_duration_us);

    Ok(WorkloadProfile {
        version: profile.version,
        captured_at: profile.captured_at,
        source_host: profile.source_host.clone(),
        pg_version: profile.pg_version.clone(),
        capture_method: "transformed".into(),
        sessions: scaled_sessions,
        metadata: Metadata {
            total_queries,
            total_sessions,
            capture_duration_us: max_offset,
        },
    })
}
```

```rust
        if !status.is_success() {
            let err_msg = resp_json["error"]["message"].as_str().unwrap_or("Unknown error");

        }

        // Extract tool use result
        let content = resp_json["content"].as_array()
            .ok_or_else(|| anyhow::anyhow!("No content in response"))?;

        for block in content {
            if block["type"] == "tool_use" {
                let input = &block["input"];
                let plan: TransformPlan = serde_json::from_value(input.clone())?;
                return Ok(plan);
            }
        }

        anyhow::bail!("No tool_use block in Claude response")
    }
```

```rust
fn cmd_transform(args: pg_retest::cli::TransformArgs) -> Result<()> {
    use pg_retest::cli::TransformAction;
    use pg_retest::profile::io;
    use pg_retest::transform::analyze::analyze_workload;

    match args.action {
        TransformAction::Analyze { workload, json } => {
            let profile = io::read_profile(&workload)?;
            let analysis = analyze_workload(&profile);

            if json {
                println!("{}", serde_json::to_string_pretty(&analysis)?);
            } else {
                println!("Workload Analysis");
                println!("=================");
                println!("  Queries: {}  Sessions: {}  Duration: {:.1}s",
                    analysis.profile_summary.total_queries,
                    analysis.profile_summary.total_sessions,
                    analysis.profile_summary.capture_duration_s,
                );
                println!();
                println!("Query Groups ({} identified, {} ungrouped):",
                    analysis.query_groups.len(), analysis.ungrouped_queries);
                for group in &analysis.query_groups {
                    println!();
                    println!("  Group {}: {} queries ({:.1}%)",
                        group.id, group.query_count, group.pct_of_total);
                    println!("    Tables: {}", group.tables.join(", "));
                    println!("    Sessions: {:?}", group.sessions);
                    println!("    Avg latency: {}us", group.avg_duration_us);
                    if !group.parameter_patterns.common_filters.is_empty() {
                        println!("    Filters: {}", group.parameter_patterns.common_filters.join(", "));
                    }
                    println!("    Sample queries:");
                    for sq in &group.sample_queries {
                        println!("      - {sq}");
                    }
                }
            }
            Ok(())
        }

        TransformAction::Plan { workload, prompt, provider, api_key, api_url, model, output, dry_run } => {
            let profile = io::read_profile(&workload)?;
            let analysis = analyze_workload(&profile);

            if dry_run {
                println!("Dry run — showing what AI would receive:");
                println!();
                println!("{}", serde_json::to_string_pretty(&analysis)?);
                return Ok(());
            }

            let api_key = api_key
                .or_else(|| std::env::var("OPENAI_API_KEY").ok())
                .ok_or_else(|| anyhow::anyhow!("API key required. Use --api-key or set ANTHROPIC_API_KEY/OPENAI_API_KEY env var"))?;

            use pg_retest::transform::planner::{create_planner, LlmProvider, PlannerConfig};
            let provider = LlmProvider::from_str(&provider)?;
            let planner = create_planner(PlannerConfig { provider, api_key, api_url, model });

            println!("Generating transform plan with {}...", planner.name());

            let rt = tokio::runtime::Runtime::new()?;
            let plan = rt.block_on(planner.generate_plan(&analysis, &prompt))?;

            let toml_str = toml::to_string_pretty(&plan)?;
            std::fs::write(&output, &toml_str)?;
            println!("Plan written to {}", output.display());
            println!();
            println!("  Groups: {}", plan.groups.len());
            println!("  Transforms: {}", plan.transforms.len());
            println!();
            println!("Review the plan, then apply:");
            println!("  pg-retest transform apply --workload {} --plan {}", workload.display(), output.display());
            Ok(())
        }

        TransformAction::Apply { workload, plan, output, seed } => {
            let profile = io::read_profile(&workload)?;
            let plan_str = std::fs::read_to_string(&plan)?;
            let transform_plan: pg_retest::transform::plan::TransformPlan = toml::from_str(&plan_str)?;

            println!("Applying transform plan...");
            println!("  Groups: {}", transform_plan.groups.len());
            println!("  Transforms: {}", transform_plan.transforms.len());

            let result = pg_retest::transform::engine::apply_transform(&profile, &transform_plan, seed)?;

            println!("  Result: {} sessions, {} queries (was: {} sessions, {} queries)",
                result.metadata.total_sessions, result.metadata.total_queries,
                profile.metadata.total_sessions, profile.metadata.total_queries);

            io::write_profile(&output, &result)?;
            println!("Wrote transformed workload to {}", output.display());
            Ok(())
        }
    }
}
```

```rust
        classification: None,
        created_at: None,
    };

        "workload_id": new_id,
        "total_sessions": result.metadata.total_sessions,
```
